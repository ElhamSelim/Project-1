#Diferences between using CONTINUE and BREAK in Loops

#Definitions of fruits in an array
fruits = ['Banana','Apple','red','Pineapple','Strawberry','Grape']
#continue --> skip the loop and continue 
for fruit in fruits:
  if fruit == 'red':
    continue
  print(fruit)
#break --> and exist the loop
for fruit in fruits:
  if fruit == 'red':
    break
  print(fruit)
